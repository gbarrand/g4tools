75  tree* prev = &m_top;
76  {std::string s;

80  new tree(prev,s);

86  if(c==',') {
87  if(s.size()) {

90  }

92  tree* _tree = new tree(prev,s);

97  if(s.size()) {
98  new tree(prev,s);
99  s.clear();
100  }
101  prev = prev->m_parent;
102  if(!prev) return false; //should not happen.
103  } else {
104  s += c;
105  }
106  }
107  }}
108  return true;
109  }
