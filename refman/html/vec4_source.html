155  // If true, then : a_vec = a_factor * this.
156  a_factor = zero();
157  bool first = true;
158  T* tp = (T*)m_data;

160  for(unsigned int i=0;i<4;i++,tp++,ap++) {
161  if( ((*tp)==zero()) && ((*ap)==zero())) {
162  continue;
163  } else if( ((*tp)!=zero()) && ((*ap)==zero())) {
164  return false;
165  } else if( ((*tp)==zero()) && ((*ap)!=zero())) {
166  return false;
167  } else {
168  if(first) {
169  a_factor = (*ap)/(*tp);
170  first = false;
171  } else {
172  if((*ap)!=(*tp)*a_factor) return false;
173  }
174  }
175  }
