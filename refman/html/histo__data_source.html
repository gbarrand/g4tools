17 inline unsigned int dim_planes(unsigned int a_dim) {
18  // m_dim = 1 -> 0
19  // m_dim = 2 -> 0+1=1
20  // m_dim = 3 -> 0+1+2=3
21  // m_dim = 4 -> 0+1+2+3=6
22  typedef unsigned int dim_t;
23  dim_t n = 0;
24  for(dim_t i=0;i<a_dim;i++) n += i;
25  return n;
26 }
